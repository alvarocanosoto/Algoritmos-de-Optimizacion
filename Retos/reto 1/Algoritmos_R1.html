#Sucesin_de_Fibonacci

def fibonacci_recursivo(n):
    if n <= 1:
        return n
    return fibonacci_recursivo(n - 1) + fibonacci_recursivo(n - 2)

def fibonacci_iterativo(n):
    if n <= 1:
        return n
    f0, f1 = 0, 1
    for _ in range(2, n + 1):
        f0, f1 = f1, f0 + f1
    return f1

# Tiempo de cmputo para ambos algoritmos
n_valores = list(range(1, 31))
tiempos_recursivo = []
tiempos_iterativo = []

for n in n_valores:
    # Tiempos para el algoritmo recursivo
    t_inicio = time.time()
    fibonacci_recursivo(n)
    t_fin = time.time()
    tiempos_recursivo.append(t_fin - t_inicio)

    # Tiempos para el algoritmo iterativo
    t_inicio = time.time()
    fibonacci_iterativo(n)
    t_fin = time.time()
    tiempos_iterativo.append(t_fin - t_inicio)

# Mostrar grfica
plt.figure(figsize=(10, 6))

# Problema de las N-Reinas
def resolver_n_reinas(tablero, fila, soluciones):
        # Comprobar si ya estn todas las reinas colocadas
        if fila == n:
            soluciones.append(tablero[:])

        else:
            for columna in range(n):
                seguro = True
                # Comprobar si la nueva reina recibe ataque de alguna otra
                for i in range(fila):
                    # Comprobar la columna y las diagonales
                    if tablero[i] == columna or i - tablero[i] == fila - columna or i + tablero[i] == fila + columna:
                        seguro = False
                        break
                        
                # La posicin es segura si no est atacada
                if seguro:
                    tablero[fila] = columna
                    resolver_n_reinas(tablero, fila + 1, soluciones)
def n_reinas(n):
    soluciones = []
    tablero = [-1] * n
    resolver_n_reinas(tablero, 0, soluciones)
    return soluciones
